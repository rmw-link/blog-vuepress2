# Модификация quickjs для импорта rust-функций - новый взгляд на расширение Kraken

## Хранилище кодов

* [github](https://github.com/rmw-lib/quickjs-rust)
* [gitee](https://gitee.com/rmw-link/quickjs-rust)
* [gitflic](https://gitflic.ru/project/rmw-link/quickjs-rust)
* [bitbucket](https://bitbucket.org/rmw-link/quickjs-rust)

## Генезис

[Kraken](https://openkraken.com) - это высокопроизводительный движок веб-рендеринга на базе `Flutter`, который использует [quickjs](https://github.com/openkraken/kraken/tree/main/bridge/third_party/quickjs) в качестве скриптового движка.

Я хотел написать несколько расширений для Kraken, используя `rust`.

Kraken [поддерживает запись расширений с помощью `dart`](https://openkraken.com/guide/advanced/custom-js-api) .

Использование [`flutter_rust_bridge`](https://github.com/fzyzcjy/flutter_rust_bridge) `rust` и `dart`.

Объединив эти два пункта, можно сделать вывод, что писать расширения Kraken с помощью `rust` несложно.  
Однако это решение требует больших затрат производительности, поскольку `dart` вызывает `rust`, а `quickjs` вызывает `dart`.

С другой стороны, в то время как сообщество `rust` имеет [`rquickjs`](https://github.com/DelSkayn/rquickjs) такие обращения к библиотеке `quickjs` в `rust`.  
Однако они вызывают `quickjs` вместо встраивания `quickjs`и не могут быть использованы для магии `quickjs`.

В этой кодовой базе я реализовал новое решение: прямое изменение исходного кода `quickjs` для поддержки расширения `rust`.

Это универсальное решение, которое можно использовать не только для модификации Kraken, но и для всех фреймворков и библиотек, в которые встроен `quickjs`.

## Демонстрация

Код test.js выглядит следующим образом :

```js
const try_run = (func, ...args) => {
  try {
    func(...args)
  } catch (err) {
    console.log('❌', err.message)
    console.log(err.stack)
  }
};

import * as rust from 'rust'
console.log("export from rust :", Object.keys(rust))

import {
  fib,
  sleep
} from 'rust'

(async () => {

  console.log('begin sleep 2s')
  await sleep(2000);
  console.log('sleep done')

  console.log('fib(3) =', fib(3));

  console.log("try catch example :")
  try_run(fib);
  try_run(fib, '*');

})()
```

Выполнить `./quickjs/qjs test.js`, выход:

```

begin sleep 2s
sleep done
fib(3) = 6
try catch example :
❌ miss : args need 1 pass 0
    at fib (native)
    at try_run (test.js:8)
    at <anonymous> (test.js:27)

❌ not number : args position 0
    at fib (native)
    at try_run (test.js:6)
    at <anonymous> (test.js:28)
```

### Реализация fib в rust

Функция fib, импортируемая в js, взята с сайта `rust/src/export/fib.rs`, и ее код выглядит следующим образом :

```rust
use crate::js::{self, arg};
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::os::raw::c_int;

#[js]
pub fn fib(n: i64) -> i64 {
  if n <= 1 {
    return if n == 1 { 1 } else { 0 };
  }
  n + fib(n - 1)
}

#[no_mangle]
pub extern "C" fn js_fib(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, fib_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => js::val(ctx, fib(n)),
  }
}
```

В настоящее время процедурный макрос `#[js]` просто добавляет константу `fib_args_len`, которая определяет количество аргументов функции.

В будущем можно будет написать процедурный макрос `./rust_macro`, чтобы обеспечить полностью автоматический экспорт функций.

### Реализация функции сна в rust

Функция sleep, импортированная в js, взята с сайта `rust/src/export/sleep.rs`, а код выглядит следующим образом :

```rust
use crate::js::{self, arg};
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::os::raw::c_int;

#[js]
pub fn fib(n: i64) -> i64 {
  if n <= 1 {
    return if n == 1 { 1 } else { 0 };
  }
  n + fib(n - 1)
}

#[no_mangle]
pub extern "C" fn js_fib(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, fib_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => js::val(ctx, fib(n)),
  }
}
use crate::{js::arg, qjs::run};
use async_io::Timer;
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::{os::raw::c_int, time::Duration};

#[js]
pub async fn sleep(n: u64) {
  Timer::after(Duration::from_millis(n)).await;
}

#[no_mangle]
pub extern "C" fn js_sleep(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, sleep_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => run(ctx, async move {
      sleep(n as u64).await;
      Ok(())
    }),
  }
}
```

Как вы можете видеть выше, все экспортируемые функции определены в каталоге `./rust/src/export`. Этот каталог `mod.rs` автоматически создается при запуске `./rust/build.xsh`, экспортируя под него все файлы `.rs`.

### Чтение и проверка входящих параметров js

Параметры считываются и проверяются на сайте `src/js/arg.rs` с помощью следующего кода :

```rust
use crate::js::{self, arg};
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::os::raw::c_int;

#[js]
pub fn fib(n: i64) -> i64 {
  if n <= 1 {
    return if n == 1 { 1 } else { 0 };
  }
  n + fib(n - 1)
}

#[no_mangle]
pub extern "C" fn js_fib(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, fib_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => js::val(ctx, fib(n)),
  }
}
use crate::{js::arg, qjs::run};
use async_io::Timer;
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::{os::raw::c_int, time::Duration};

#[js]
pub async fn sleep(n: u64) {
  Timer::after(Duration::from_millis(n)).await;
}

#[no_mangle]
pub extern "C" fn js_sleep(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, sleep_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => run(ctx, async move {
      sleep(n as u64).await;
      Ok(())
    }),
  }
}
use crate::js::throw;

use quickjs_ffi::{JSContext, JSValue, JS_IsNumber, JS_ToInt64};
use std::{mem::MaybeUninit, os::raw::c_int};

pub(crate) fn arg_miss(ctx: *mut JSContext, argc: c_int, need: c_int) -> Result<(), JSValue> {
  if argc < need {
    throw(ctx, format!("miss : args need {need} pass {argc}"))?
  }
  Ok(())
}

pub(crate) fn arg_i64(ctx: *mut JSContext, argv: *mut JSValue, pos: isize) -> Result<i64, JSValue> {
  unsafe {
    let val = *argv.offset(pos);
    if JS_IsNumber(val) == 0 {
      throw(ctx, format!("not number : args position {pos}"))?
    }
    let mut n = MaybeUninit::uninit();
    JS_ToInt64(ctx, n.as_mut_ptr() as _, val);
    Ok(n.assume_init())
  }
}
```

В настоящее время проверяется только количество аргументов и считывается тип i64.

Вы можете добавить эти функции по мере необходимости, см. функции в [qjs_sys](https://docs.rs/qjs-sys/0.1.2/qjs_sys/), начиная с `JS_To`.

  None,
  Undefined,
  I64(i64),
  CString(CString),
}

impl From<()> for Val {
  fn from(_: ()) -> Self {
    Val::Undefined
  }
}

impl From<i64> for Val {
  fn from(t: i64) -> Self {
    Val::I64(t)
  }
}

impl From<CString> for Val {
  fn from(t: CString) -> Self {
    Val::CString(t)
  }
}

pub(crate) fn val(ctx: *mut JSContext, t: impl Into<Val>) -> JSValue {
  match t.into() {
    Val::None => JS_NULL,
    Val::Undefined => JS_UNDEFINED,
    Val::I64(n) => unsafe { JS_NewInt64(ctx, n) },
    Val::CString(cstr) => unsafe { JS_NewString(ctx, cstr.as_ptr()) },

Для преобразования из `None`, `()`, `i64`и CString в `js` определены только четыре типа. Вы можете добавить столько, сколько захотите.

Другие типы данных могут быть объявлены в функциях в [qjs_sys](https://docs.rs/qjs-sys/0.1.2/qjs_sys/), начиная с `JS_New`.

## Среда разработки

Я разрабатываю на ноутбуке Apple, rust использует 1.62.0-nightly.

Сначала установите [direnv](https://direnv.net), перейдите в каталог и `direnv allow` на некоторое время.

Установите python3, затем `pip3 install -r ./requirements.txt`

запустить `./build.xsh` для компиляции и запуска демонстрационной программы

По умолчанию официальный репозиторий quickjs будет клонирован, если вы хотите изменить quickjs в репозитории Kraken, сначала

`git clone --recursive git@github.com:openkraken/kraken.git --depth=1`

тогда сделайте следующее

```bash
rm -rf quickjs
ln -s ../kraken/bridge/third_party/quickjs .
```

Наконец, повторно запустите `./build.xsh`

## Структура каталога

* `./quickjs_rust`  
  Модификация c-файла кода quickjs
  
* `./quickjs_ffi`  
  Экспортируйте функции из заголовочного файла `quickjs` в `rust`
  
* `./rust`  
  Используйте `rust` для реализации функций в `quickjs`.
  
  * `./rust/src/qjs.rs`  
    Реализация асинхронных вызовов. Поскольку `quickjs` является однопоточным, вызовы функций с участием `quckjs` записываются в главном потоке.
* `./rust_macro`  
  `rust` Реализация макроса процедуры `#[js]`
  
  В будущем, см. [wasmedge-quickjs](https://github.com/second-state/wasmedge-quickjs/blob/70efe8520dc65636cb81b7225e2a6dae47cfad49/src/quickjs_sys/mod.rs#L122) для автоматического экспорта rust-функций в js-функции. [wasmedge-quickjs → JsFunctionTrampoline](https://github.com/second-state/wasmedge-quickjs/blob/70efe8520dc65636cb81b7225e2a6dae47cfad49/src/quickjs_sys/mod.rs#L122)
  

## Создание скриптов `build.xsh`

Без лишних слов перейдем непосредственно к исходному коду сценария сборки `build.xsh`.

```xonsh
#!/usr/bin/env xonsh

from pathlib import Path
from os.path import dirname,abspath,exists,join
PWD = dirname(abspath(__file__))
cd @(PWD)

p".xonshrc".exists() && source .xonshrc

quickjs = 'quickjs'

if not exists(quickjs):
  git clone git@github.com:bellard/@(quickjs).git --depth=1

./quickjs_rust/patch.py

./rust/build.xsh
./quickjs_rust/gen.py

def ln_s(li):
  for arg in li.split(' '):
    fp = join(quickjs,arg)
    if not exists(fp):
      ln -s @(PWD)/@(arg) @(fp)

ln_s('quickjs_rust rust quickjs_ffi rust_macro')

cd @(quickjs)
make qjs

cd @(PWD)
./quickjs/qjs --unhandled-rejection -m test.js 2>&1 | tee test.js.out
```

## Объяснение принципа

### `quickjs_rust/patch.py`

Запуск `./quickjs_rust/patch.py` внесет некоторые незначительные изменения в исходный код `quickjs`.

Одна из функций `JS_AddRust` используется для инъекции в модуль rust.

`rust_run` инжектируется в `JS_ExecutePendingJob` для вызова асинхронных функций.

Скриншот всех изменений показан ниже:

![](https://raw.githubusercontent.com/gcxfd/img/gh-pages/ep2Xgg.png)

### `quickjs_rust.h`

Из приведенных выше изменений видно, что мы ввели новый заголовочный файл `quickjs_rust.h` со следующим кодом

```c
#ifndef QUICKJS_RUST_H
#define QUICKJS_RUST_H

#include "../quickjs/quickjs.h"
#include "../rust/rust.h"

#define countof(x) (sizeof(x) / sizeof((x)[0]))
#define JS_RUSTFUNC_DEF(name) JS_CFUNC_DEF(#name, name##_args_len, js_##name)
#include "./js_rust_funcs.h"

static const unsigned int js_rust_funcs_count = countof(js_rust_funcs);

static int
js_rust_init(JSContext* ctx, JSModuleDef* m)
{
  return JS_SetModuleExportList(ctx, m, js_rust_funcs,
      js_rust_funcs_count);
}

#define JS_INIT_MODULE js_init_module_rust

JSModuleDef* JS_INIT_MODULE(JSContext* ctx, const char* module_name)
{
  JSModuleDef* m;
  m = JS_NewCModule(ctx, module_name, js_rust_init);
  if (!m)
    return NULL;
  js_rust_init(ctx, m);
  return m;
}

void JS_AddRust(JSContext* ctx, JSRuntime* rt)
{
  JSModuleDef* m = JS_INIT_MODULE(ctx, "rust");
  for (unsigned int i = 0; i < js_rust_funcs_count; i++) {
    JS_AddModuleExport(ctx, m, js_rust_funcs[i].name);
  }
  rust_init(ctx, rt);
}

#endif
```

### `rust/rust.h`

Вы можете видеть, что `quickjs_rust/quickjs_rust.h` вводит `quickjs_rust/js_rust_funcs.h`, который автоматически генерируется из заголовочного файла функции экспорта rust `rust/rust.h` и не должен быть изменен вручную.

А `rust/rust.h` создается путем вызова cbindgen из `./rust/build.xsh`.

### `rust/build.xsh`

```xonsh
#!/usr/bin/env xonsh

from os.path import dirname,abspath
import platform
PWD = dirname(abspath(__file__))
cd @(PWD)

p"../.xonshrc".exists() && source ../.xonshrc

./src/export/mod.gen.py

system = platform.system().lower()
if system == 'darwin':
  system = f'apple-{system}'

TARGET=f'{platform.machine()}-{system}'

def cbindgen():
  cbindgen -q --config cbindgen.toml --crate rust --output rust.h

try:
  cbindgen()
except:
  cargo clean
  cbindgen()

cargo build \
--release \
-Z build-std=std,panic_abort \
-Z build-std-features=panic_immediate_abort \
--target @(TARGET)

mv ./target/@(TARGET)/release/librust.a ./target/release
```
